(
    ID SERIAL PRIMARY KEY,
    ISSUESLISTS_ID BIGINT UNSIGNED NOT NULL,

    DESCRIPTION VARCHAR(1000),
    USER_ID_ASSIGNEDTO_ID BIGINT UNSIGNED NOT NULL,
    FOREIGN KEY (USER_ID_ASSIGNEDTO_ID) REFERENCES USERS(ID)
);
